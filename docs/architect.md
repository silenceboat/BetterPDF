import sys

from PySide6.QtWidgets import (

    QScrollArea, QFileDialog, QToolBar

from PySide6.QtCore import Qt


class PDFViewer(QMainWindow):
    def __init__(self):
        super().__init__()
        self.setWindowTitle("DeepRead AI")
        self.resize(1200, 800)
        
        # å·¥å…·æ 
        toolbar = QToolBar()
        self.addToolBar(toolbar)
        open_action = QAction("ğŸ“‚ æ‰“å¼€PDF", self)
        open_action.triggered.connect(self.open_file)
        toolbar.addAction(open_action)
        
        # æ»šåŠ¨åŒºåŸŸ
        self.scroll = QScrollArea()
        self.label = QLabel()
        self.label.setAlignment(Qt.AlignmentFlag.AlignCenter)
        self.scroll.setWidget(self.label)
        self.scroll.setWidgetResizable(True)
        self.setCentralWidget(self.scroll)
        
        self.doc = None
        self.current_page = 0
    
    def open_file(self):

            self, "é€‰æ‹©PDF", "", "PDF Files (*.pdf)"

        if path:
            self.doc = fitz.open(path)
            self.render_page(0)
    
    def render_page(self, page_num):
        page = self.doc[page_num]
        pix = page.get_pixmap(matrix=fitz.Matrix(2, 2))
        img = QImage(pix.samples, pix.width, pix.height,
                     pix.stride, QImage.Format.Format_RGB888)
        self.label.setPixmap(QPixmap.fromImage(img))
        self.setWindowTitle(
            f"DeepRead AI â€” ç¬¬{page_num+1}/{len(self.doc)}é¡µ"
        )
    
    def keyPressEvent(self, event):
        if self.doc:
            if event.key() == Qt.Key.Key_Right:
                self.current_page = min(
                    self.current_page + 1, len(self.doc) - 1
                )
                self.render_page(self.current_page)
            elif event.key() == Qt.Key.Key_Left:
                self.current_page = max(self.current_page - 1, 0)
                self.render_page(self.current_page)


if __name__ == "__main__":
